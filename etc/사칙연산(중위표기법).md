# 사칙연산(중위표기법)

```javascript
function solution() {
  let answer = ["10", "+", "5", "*", "9"];
  let tempArr = [];
  for (let i = 0; i < answer.length; i++) {
    if (tempArr.length === 2) {
      let operator = tempArr.pop();
      let num = Number(tempArr.pop());

      switch (operator) {
        case "*":
          tempArr.push(num * Number(answer[i]));
          break;
        case "/":
          tempArr.push(num / Number(answer[i]));
          break;
        case "+":
          tempArr.push(num + Number(answer[i]));
          break;
        case "-":
          tempArr.push(num - Number(answer[i]));
          break;
      }
    } else {
      tempArr.push(answer[i]);
    }
  }
}

solution();
```
